# RK3588 CLI (RKNN + ROS2)
# 当前约定：无量纲网格（grid units），OccupancyGrid resolution 固定名义值 1.0（非物理米制）

# 0) 进入工程根目录
cd "$(git rev-parse --show-toplevel 2>/dev/null || pwd)"

# 1) 编译
source /opt/ros/humble/setup.bash
rm -rf build_ninja

cmake -S . -B build_ninja -G Ninja \
  -DCMAKE_BUILD_TYPE=Release \
  -DYOSEG_WITH_ROS2=ON \
  -DUSE_RKNN=ON
  
cmake --build build_ninja -j$(nproc)



# 2) 终端1：发布（固定 RKNN）
source /opt/ros/humble/setup.bash
./build_ninja/apps/realtime_runner/realtime_runner \
  --source 0 \
  --backend rknn \
  --model ./weights/0414_qy++.rknn \
  --frames 1000 \
  --warmup-frames 8 \
  --queue-capacity 4 \
  --net-w 640 \
  --net-h 640 \
  --ros-enable \
  --ros-rate 5 \
  --ros-occ-topic /octomap/occupancy \
  --ros-cloud-topic /octomap/points \
  --profile-out ./profile_ros_rknn.csv

# 说明：--ros-cell-size 为兼容参数，当前模式下会被忽略

# 3) 终端2：话题检查
source /opt/ros/humble/setup.bash
ros2 topic list | grep octomap
ros2 topic echo /octomap/occupancy --once
ros2 topic echo /octomap/points --once
ros2 topic hz /octomap/occupancy
ros2 topic hz /octomap/points

# 4) 终端3：RViz2（可选）
source /opt/ros/humble/setup.bash
rviz2
